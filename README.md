src/
├── app/                    # 1. Инициализация приложения
│   ├── providers/          # Провайдеры (Store, Router, Theme и т.д.)
│   ├── styles/             # Глобальные стили (reset.css, variables)
│   └── main.tsx/js         # Точка входа
│
├── pages/                  # 2. Страницы приложения (композиция виджетов)
│   ├── leagues/            # Страница со списком лиг
│   ├── teams/              # Страница со списком команд
│   ├── league-calendar/    # Календарь конкретной лиги
│   └── team-calendar/      # Календарь конкретной команды
│
├── widgets/                # 3. Крупные самостоятельные блоки
│   ├── header/             # Шапка сайта (включает поиск и навигацию)
│   ├── league-list/        # Список лиг (включает карточки лиг и пагинацию)
│   ├── team-list/          # Список команд (включает карточки команд и пагинацию)
│   └── match-table/        # Таблица матчей (включает фильтр по датам и строки матчей)
│
├── features/               # 4. Бизнес-логика и взаимодействия пользователя
│   ├── search/             # Логика текстового поиска (UI + обработка)
│   ├── date-filter/        # Фильтр матчей по датам "С" и "По"
│   └── pagination/         # Логика переключения страниц
│
├── entities/               # 5. Бизнес-сущности (данные и UI для отображения сущности)
│   ├── league/             # Сущность "Лига" (UI карточки, типы, API запросы для лиги)
│   ├── team/               # Сущность "Команда" (UI карточки, типы, API запросы для команды)
│   └── match/              # Сущность "Матч" (UI строки таблицы, форматирование статусов/счета)
│
└── shared/                 # 6. Переиспользуемый инфраструктурный код
    ├── api/                # Базовый инстанс API (настройка axios/fetch, базовый URL, токен)
    ├── ui/                 # UI Kit: базовые "глупые" компоненты (Button, Input, Card, Spinner)
    ├── lib/                # Утилиты (например, функция конвертации UTC в локальное время)
    └── config/             # Конфигурации (переменные окружения, константы роутов)
